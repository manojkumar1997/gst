CREATE TABLE BASE.M_EXCHANGE_RATE(
  PRODUCT_ID           TINYINT UNSIGNED AUTO_INCREMENT,
  PRODUCT_NAME 			VARCHAR(4),
  PRODUCT_PRICE 			VARCHAR(4),
  PRODUCT_GST 			  DECIMAL(10,2) DEFAULT 0,
  CREATED_DT			  DATETIME,
  UPDATED_DT     		DATETIME,
  PRIMARY KEY(PRODUCT_ID),
  CONSTRAINT UK1_EXCHANGE_RATE UNIQUE (PRODUCT_ID,PRODUCT_NAME)
);

CREATE INDEX IDX_M_EXCHANGE_RATE ON M_EXCHANGE_RATE(PRODUCT_NAME);
